# --- ETAPA 1: Compilación de Frontend (Vue) ---
FROM node:20-alpine AS build-stage
WORKDIR /app
# Copiamos los archivos locales al contenedor
COPY . .
RUN npm install
RUN npm run build

# --- ETAPA 2: Servidor Web (PHP 8.2 + Apache) ---
FROM php:8.2-apache

# Instalar dependencias del sistema y módulos PHP necesarios
RUN apt-get update && apt-get install -y \
    libpng-dev libjpeg-dev libfreetype6-dev zip libzip-dev \
    unzip git ssl-cert libpq-dev \
    && docker-php-ext-configure gd --with-freetype --with-jpeg \
    && docker-php-ext-install gd pdo pdo_mysql pdo_pgsql zip \
    && a2enmod rewrite ssl

# Configurar Apache y PHP
COPY ./docker-config/000-default.conf /etc/apache2/sites-available/000-default.conf
COPY ./docker-config/php.ini /usr/local/etc/php/conf.d/app-php.ini

# Copiar el proyecto desde la etapa de build
COPY --from=build-stage /app /var/www/html

# Instalar Composer y dependencias de PHP
COPY --from=composer:latest /usr/bin/composer /usr/bin/composer
ENV COMPOSER_ALLOW_SUPERUSER=1
RUN composer install --no-dev --optimize-autoloader

# REQUISITO: Permisos correctos para Laravel
RUN chown -R www-data:www-data /var/www/html/storage /var/www/html/bootstrap/cache
RUN chmod -R 775 /var/www/html/storage

# Exponer puertos para HTTP y HTTPS
EXPOSE 80 443

CMD ["apache2-foreground"]
